Algo to swap two integers : 

Logic 1 : 
	temp = a;

import java.util.*;
class Demo{
	void swap(int a,int b){
		int temp;
		temp = a;
		a = b;
		b = temp;
	}
}
class Test{
	public static void main(String args[]){
		System.out.println(Demo.swap(10,20));
	}
}
